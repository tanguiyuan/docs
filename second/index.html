  • 人脸特征提取

    QString Face::RecognizeFace(cv::Mat inimg, QRect &rect)
{

  IplImage imgTmp = inimg;//虹软支持IplImage

  IplImage *originalImg=cvCloneImage(&imgTmp);
  //图像裁剪，宽度做四字节对齐，若能保证图像是四字节对齐这步可以不用做
  IplImage* img= cvCreateImage(cvSize(originalImg->width - originalImg->width % 4, originalImg->height), IPL_DEPTH_8U, originalImg->nChannels);
  CutIplImage(originalImg, img, 0, 0);
  ////图像数据以结构体形式传入，对更高精度的图像兼容性更好
  ASVLOFFSCREEN offscreen = { 0 };
  offscreen.u32PixelArrayFormat = ASVL_PAF_RGB24_B8G8R8;
  offscreen.i32Width = img->width;
  offscreen.i32Height = img->height;
  offscreen.pi32Pitch[0] = img->widthStep;
  offscreen.ppu8Plane[0] = (MUInt8*)img->imageData;

  ASF_MultiFaceInfo detectedFaces = { 0 };
  ASF_FaceFeature feature = { 0 };
  if (img)
  {

       res = ASFDetectFacesEx(handle, &offscreen, &detectedFaces);
       //提取图像数据,在这里offscreen和detectedFaces都有，但是没有handle后面初始化引擎会说
      if (MOK != res)//打印失败原因
      {
          qDebug() << "ASFDetectFacesEx failed: " << res;

      }
      else//成功
      {
          // 打印人脸检测结果

          for (int i = 0; i < detectedFaces.faceNum; i++)
          {

                  rect.setLeft(detectedFaces.faceRect[i].left);
                  rect.setTop(detectedFaces.faceRect[i].top);
                  rect.setRight(detectedFaces.faceRect[i].right);
                  rect.setBottom(detectedFaces.faceRect[i].bottom);//返回人脸框数据
                  //这里只写了一个人脸框，太累了懒得改成多框
                  //这里取检测到的所有人脸进行特征提取
                  ASF_SingleFaceInfo singleDetectedFaces = { 0 };
                  singleDetectedFaces.faceRect.left = detectedFaces.faceRect[i].left;
                  singleDetectedFaces.faceRect.top = detectedFaces.faceRect[i].top;
                  singleDetectedFaces.faceRect.right = detectedFaces.faceRect[i].right;
                  singleDetectedFaces.faceRect.bottom = detectedFaces.faceRect[i].bottom;
                  singleDetectedFaces.faceOrient = detectedFaces.faceOrient[i];
                  //提取数据
                  res = ASFFaceFeatureExtract(handle, img->width, img->height, ASVL_PAF_RGB24_B8G8R8, (MUInt8*)img->imageData,
                      &singleDetectedFaces, &feature);

                  if (MOK != res)//失败
                  {
                      qDebug() <<"ASFFaceFeatureExtract failed: %d\n"<<res;
                  }//调用face里面的feather_compare类
                  //实现人脸识别，不要以为做了人脸检测 哈哈哈哈
                  feather_compare(feature);

          }


      }

      //释放图像内存，这里只是做人脸检测，若还需要做特征提取等处理，图像数据没必要释放这么早
      cvReleaseImage(&img);
  }
  cvReleaseImage(&originalImg);



  return "ok";

  }

    Face::feather_compare(ASF_FaceFeature feature)
//传入当前人脸特征
{
  QSqlDatabase db = QSqlDatabase::addDatabase("QSQLITE");
  db.setDatabaseName("./feathers.db");//打开数据库
  db.open();
  QSqlQuery tmpQuery(db);
  QSqlQuery query = tmpQuery;
  query.exec("SELECT * FROM features");
  if(!query.exec())
  {
     // qDebug() << "Error: " << query.lastError();
  }
  else
  {
      qDebug() << "select success";
  }
     
  while(query.next())//取出所有数据
      {
      int id=query.value(0).toInt();
      QString name=query.value(1).toString();
      QByteArray byte=query.value(2).toByteArray();
      ASF_FaceFeature Feature;
      //数据库中的数据转换成人类数据结构体需要的数据。
      Feature.feature=reinterpret_cast< MByte*>(byte.data());
      Feature.featureSize=byte.size();
      copyfeature1.featureSize = Feature.featureSize;
      copyfeature1.feature = reinterpret_cast<MByte*>(malloc(static_cast<size_t>(Feature.featureSize)));
      memset(copyfeature1.feature, 0, static_cast<size_t>(Feature.featureSize));
      memcpy(copyfeature1.feature, Feature.feature, static_cast<size_t>(Feature.featureSize));

      MFloat confidenceLevel;//阈值
      //传入的人脸识别数据feature 和copyfeature1比较
      res = ASFFaceFeatureCompare(handle, &feature, &copyfeature1, &confidenceLevel);//比较
      qDebug() <<"Compare sucessed: "<<confidenceLevel;
      if(confidenceLevel>0.8)//如果阈值大于0.8就是识别成功
      {
          qDebug() <<"name:"<<name;
          if(!onlinelist.contains(name))//这些不用管，是其他作用
            {
             onlinelist.append(name);
            }

      }

      };

  db.close();
 }

  • 初始化引擎

    使用上面三个功能最开始需要初始化引擎

    Face::Init()
{

  //初始化引擎库,在.h文件MHandle handle ;
  
  handle = NULL;
  MInt32 nScale = 32;
  MInt32 faceNum = 5;
  MInt32 mask = ASF_FACE_DETECT | ASF_FACERECOGNITION ;
  //初始化引擎才能使用
  res = ASFInitEngine(ASF_DETECT_MODE_VIDEO, ASF_OP_0_ONLY, nScale, faceNum, mask, &handle);
  if (res != MOK)
  {
      qDebug() << "ASFInitEngine failed: " << res;

  }
  else
  {
      qDebug() << "ASFInitEngine sucessed: " << res;

  }
  DataBase data;
  data.Init_sql("./feathers.db");

}
